/** * ============================================ * JSAPI 收银台 · 支付宝小程序页面 * ============================================ * * 支付流程： * 1. 输入金额 → 点"立即支付" * 2. my.getAuthCode() 获取用户授权码 * 3. 后端用 authCode 换 user_id → alipay.trade.create → tradeNO * 4. my.tradePay({ tradeNO }) 唤起原生支付 * 5. 支付成功回调 → 显示结果 */ const app = getApp(); Page({ data: { amount: '', subject: '商品购买', paying: false, btnText: '', statusText: '', statusIcon: '', statusType: '', // success | error | info }, // ── 金额输入 ── onAmountInput(e) { // 只允许数字和小数点，最多两位小数 let val = e.detail.value; val = val.replace(/[^\d.]/g, ''); const parts = val.split('.'); if (parts.length > 2) val = parts[0] + '.' + parts.slice(1).join(''); if (parts[1] && parts[1].length > 2) val = parts[0] + '.' + parts[1].slice(0, 2); this.setData({ amount: val }); }, // ── 商品说明输入 ── onSubjectInput(e) { this.setData({ subject: e.detail.value }); }, // ── 发起支付 ── async handlePay() { const amount = (this.data.amount || '').trim(); const subject = (this.data.subject || '商品购买').trim(); // 校验金额 if (!amount || isNaN(parseFloat(amount)) || parseFloat(amount) <= 0) { my.showToast({ content: '请输入有效金额', type: 'fail', duration: 2000 }); return; } this.setData({ paying: true, btnText: '获取授权中...', statusText: '', statusIcon: '', statusType: '', }); let authCode; try { // ═══ Step 1: 获取授权码 ═══ authCode = await this.getAuthCode(); console.log('[收银台] 获取 authCode 成功'); // ═══ Step 2: 后端创建交易 ═══ this.setData({ btnText: '创建订单中...' }); const createResp = await this.request('/cashier/create', { amount, subject, auth_code: authCode, }); if (createResp.code !== 'OK') { throw new Error(createResp.message || '创建订单失败'); } const tradeNO = createResp.trade_no; console.log('[收银台] 交易创建成功, tradeNO:', tradeNO); // ═══ Step 3: 唤起原生支付 ═══ this.setData({ btnText: '唤起支付...' }); const payResult = await this.tradePay(tradeNO); console.log('[收银台] 支付结果:', JSON.stringify(payResult)); // ═══ Step 4: 处理结果 ═══ if (payResult.resultCode === '9000') { // 9000 = 支付成功 this.setData({ paying: false, btnText: '', statusText: `支付成功！¥${parseFloat(amount).toFixed(2)}`, statusIcon: '✓', statusType: 'success', }); my.vibrate({ type: 'medium' }); } else if (payResult.resultCode === '8000') { // 8000 = 正在处理中（结果未知，需查询） this.setData({ paying: false, btnText: '', statusText: '支付处理中，请稍后查看结果', statusIcon: '⏳', statusType: 'info', }); } else { // 其他 = 支付失败 / 用户取消 const errMap = { '6001': '用户取消支付', '6002': '网络连接失败', '6004': '支付结果未知', }; const msg = errMap[payResult.resultCode] || `支付失败 (${payResult.resultCode})`; this.setData({ paying: false, btnText: '', statusText: msg, statusIcon: '✗', statusType: 'error', }); } } catch (err) { console.error('[收银台] 支付异常:', err); my.showToast({ content: err.message || '支付服务异常', type: 'fail', duration: 3000, }); this.setData({ paying: false, btnText: '', statusText: err.message || '支付失败', statusIcon: '✗', statusType: 'error', }); } }, // ── 获取授权码 ── getAuthCode() { return new Promise((resolve, reject) => { my.getAuthCode({ scopes: ['auth_user'], success: (res) => { if (res.authCode) { resolve(res.authCode); } else { reject(new Error('未获取到授权码')); } }, fail: (err) => { console.error('[收银台] getAuthCode 失败:', JSON.stringify(err)); reject(new Error(err.errorMessage || '获取支付宝授权失败')); }, }); }); }, // ── 发起交易创建请求 ── request(path, data) { const url = app.globalData.apiBase + path; const bodyStr = JSON.stringify(data); console.log('[收银台] request ->', url, 'body:', bodyStr); return new Promise((resolve, reject) => { my.request({ url, method: 'POST', headers: { 'Content-Type': 'application/json' }, data, timeout: 20000, success: (res) => { console.log('[收银台] request success, status:', res.status, 'data:', JSON.stringify(res.data)); if (res.status === 200) { resolve(res.data); } else { const msg = (res.data && res.data.message) || (res.data && res.data.msg) || `服务器错误 (${res.status})`; reject(new Error(msg)); } }, fail: (err) => { console.error('[收银台] request fail:', JSON.stringify(err)); const msg = err.errorMessage || err.error || '网络请求失败，请检查网络'; reject(new Error(msg)); }, }); }); }, // ── 唤起支付 ── tradePay(tradeNO) { return new Promise((resolve, reject) => { my.tradePay({ tradeNO, success: (res) => { resolve(res); }, fail: (err) => { console.error('[收银台] tradePay 失败:', JSON.stringify(err)); reject(new Error(err.errorMessage || '唤起支付失败')); }, }); }); }, });
